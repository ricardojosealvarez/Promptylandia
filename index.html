function esc(s){ return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;'); }

function tok(t){ return t.toLowerCase().trim().split(/\s+/).filter(x=>x.length>1); }

function hl(text,tokens){

if(!tokens.length) return esc(text);

let r=esc(text);

tokens.forEach(t=>{

const re=new RegExp('('+t.replace(/[.*+?^${}()|[\]\\]/g,'\\$&')+')','gi');

r=r.replace(re,'<mark>$1</mark>');

});

return r;

}

async function sbFetch(path, opts={}, useService=false){

const key = useService ? SUPABASE_SERVICE : SUPABASE_ANON;

const res = await fetch(SUPABASE_URL+'/rest/v1/'+path, {

...opts,

headers:{

'apikey': key,

'Authorization': 'Bearer '+key,

'Content-Type': 'application/json',

'Prefer': opts.prefer||'',

...(opts.headers||{})

}

});

if(!res.ok){

const txt = await res.text();

throw new Error(res.status+': '+txt);

}

const ct = res.headers.get('content-type')||'';

if(ct.includes('json') && res.status!==204) return res.json();

return null;

}

let h='<button class="pb" onclick="'+callbackName+'('+(current-1)+')" '+(current===1?'disabled':'')+'>&lsaquo; Ant</button>';

const range=[];

for(let i=1;i<=total;i++){

if(i===1||i===total||(i>=current-2&&i<=current+2)) range.push(i);

else if(range[range.length-1]!=='…') range.push('…');

}

range.forEach(p=>{

if(p==='…') h+='<span class="pi">…</span>';

else h+='<button class="pb '+(p===current?'active':'')+'" onclick="'+callbackName+'('+p+')">'+p+'</button>';

});

h+='<button class="pb" onclick="'+callbackName+'('+(current+1)+')" '+(current===total?'disabled':'')+'>Sig &rsaquo;</button>';

el.innerHTML=h;

}

// ── IMPORT EXCEL ──────────────────────────────────────────────────

const REQUIRED_COLS=['CATEGORIA','SUBCATEGORIA','IA','NOMBRE','PROMPT'];

function handleImportFile(input){

const file=input.files[0]; if(!file) return;

input.value='';

const reader=new FileReader();

reader.onload=e=>{

try{ processImportBuffer(e.target.result,file.name); }

catch(err){ toast('Error leyendo fichero: '+err.message,true); }

};

reader.readAsArrayBuffer(file);

}

async function processImportBuffer(buffer,filename){

const XLSX=window.XLSX;

if(!XLSX){toast('Librería XLSX no disponible.',true);return;}

const wb=XLSX.read(buffer,{type:'array'});

const ws=wb.Sheets[wb.SheetNames[0]];

const rows=XLSX.utils.sheet_to_json(ws,{defval:''});

if(!rows.length){showImportResult(filename,0,0,[],['El fichero está vacío.']);return;}

const upperMap={};

Object.keys(rows[0]).forEach(k=>{upperMap[k.trim().toUpperCase()]=k;});

const missing=REQUIRED_COLS.filter(c=>!upperMap[c]);

if(missing.length){showImportResult(filename,0,0,[],['Faltan columnas: '+missing.join(', ')]);return;}

// Fetch existing keys for dedup

let existingKeys=new Set();

try{

let all=[], page=0, pageSize=1000, done=false;

while(!done){

const data=await sbFetch('prompts?select=nombre,categoria&order=id.asc',

{headers:{'Range':page*pageSize+'-'+(page*pageSize+pageSize-1),'Prefer':''},prefer:''},true);

if(!data||data.length===0){done=true;}

else{data.forEach(r=>existingKeys.add(r.nombre.trim().toLowerCase()+'||'+r.categoria.trim().toLowerCase()));

if(data.length<pageSize) done=true; else page++;}

}

} catch(e){toast('Error cargando existentes: '+e.message,true);return;}

let toInsert=[],skipped=[],errors=[];

rows.forEach((row,i)=>{

const rowNum=i+2;

const cat=String(row[upperMap['CATEGORIA']]||'').trim(),

sub=String(row[upperMap['SUBCATEGORIA']]||'').trim(),

ia=String(row[upperMap['IA']]||'').trim().toUpperCase(),

nombre=String(row[upperMap['NOMBRE']]||'').trim(),

prompt=String(row[upperMap['PROMPT']]||'').trim(),

notas=upperMap['NOTAS']?String(row[upperMap['NOTAS']]||'').trim():'';

if(!cat||!sub||!ia||!nombre||!prompt){

errors.push('Fila '+rowNum+': campos vacíos');return;

}

const key=nombre.toLowerCase()+'||'+cat.toLowerCase();

if(existingKeys.has(key)){skipped.push('Fila '+rowNum+': "'+nombre+'" / '+cat);return;}

existingKeys.add(key);

toInsert.push({categoria:cat,subcategoria:sub,ia,nombre,prompt,notas});

});

// Insert in batches

let added=0;

const BATCH=100;

for(let i=0;i<toInsert.length;i+=BATCH){

const batch=toInsert.slice(i,i+BATCH);

try{

await sbFetch('prompts',{method:'POST',body:JSON.stringify(batch),prefer:'return=minimal'},true);

added+=batch.length;

} catch(e){ errors.push('Error insertando lote: '+e.message); }

}

if(added>0){

const c=parseInt(document.getElementById('totalCount').textContent.replace(/[^0-9]/g,''))||0;

document.getElementById('totalCount').textContent=(c+added).toLocaleString('es-ES');

loadEditPage(1);

}

showImportResult(filename,rows.length,added,skipped,errors);

}

function showImportResult(filename,total,added,skipped,errors){

document.getElementById('importTitle').textContent='Importación: '+filename;

document.getElementById('importSummary').innerHTML=

'<div class="import-stats">'+

'<div class="istat total"><span class="num">'+total+'</span><span class="lbl">Leídas</span></div>'+

'<div class="istat added"><span class="num">'+added+'</span><span class="lbl">Añadidas</span></div>'+

'<div class="istat skipped"><span class="num">'+skipped.length+'</span><span class="lbl">Duplicadas</span></div>'+

'<div class="istat errors"><span class="num">'+errors.length+'</span><span class="lbl">Errores</span></div>'+

'</div>';

const sd=document.getElementById('importSkippedList');

sd.style.display=skipped.length?'block':'none';

if(skipped.length) document.getElementById('importSkippedItems').innerHTML=skipped.map(s=>'<div>'+esc(s)+'</div>').join('');

const ed=document.getElementById('importErrorList');

ed.style.display=errors.length?'block':'none';

if(errors.length) document.getElementById('importErrorItems').innerHTML=errors.map(e=>'<div>'+esc(e)+'</div>').join('');

document.getElementById('importOverlay').classList.add('open');

}
